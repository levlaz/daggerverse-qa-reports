func TestHello(t *testing.T) {
    tests := []struct {
        name     string
        input    *string
        expected string
    }{
        {"default", nil, "Hello!"},
        {"with name", "World", "Hello, World!"},
    }
    
    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            result := hello(ctx, tc.input)
            if result != tc.expected {
                t.Errorf("expected %q, got %q", tc.expected, result)
            }
        })
    }
}
