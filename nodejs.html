2. Add Input Validation

Add version validation to ensure compatible versions are used:


func validateNodeVersion(version string) error {
    if version == "latest" {
        return nil
    }
    matched, _ := regexp.MatchString(`^\d+\.\d+\.\d+$`, version)
    if !matched {
        return fmt.Errorf("invalid node version format: %s", version)
    }
    return nil
}
