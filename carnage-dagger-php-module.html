Module Description

A PHP module for Dagger that provides PHP container management with support for various PHP configurations including CLI, Apache, FPM, and ZTS versions.

Functions

  • apache(version: String = "8.3", os: String = "alpine") - Creates a PHP Apache container
  • cli(version: String = "8.3", os: String = "alpine") - Creates a PHP CLI container
  • fpm(version: String = "8.3", os: String = "alpine") - Creates a PHP-FPM container
  • zts(version: String = "8.3", os: String = "alpine") - Creates a PHP ZTS (Thread Safe) container

func validatePhpVersion(version string) error {
    supportedVersions := []string{"8.3", "8.2", "8.1", "8.0"}
    for _, v := range supportedVersions {
        if v == version {
            return nil
        }
    }
    return fmt.Errorf("unsupported PHP version: %s", version)
}
