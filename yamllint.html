2. Add Detailed Error Reporting

Enhance error messages with more context and suggestions for fixing issues.

type LintError struct {
    File    string
    Line    int
    Column  int
    Message string
    Level   string
}

func (y *Yamllint) check(ctx context.Context, src *Directory) (*Directory, error) {
    results := []LintError{}
    // Process yamllint output and populate results
    
    if len(results) > 0 {
        errorReport := strings.Builder{}
        errorReport.WriteString("YAML Lint Issues Found:\n\n")
        for _, err := range results {
            fmt.Fprintf(&errorReport, 
                "File: %s\nLine: %d, Column: %d\nLevel: %s\nIssue: %s\n\n",
                err.File, err.Line, err.Column, err.Level, err.Message)
        }
        return nil, fmt.Errorf(errorReport.String())
    }
    return src, nil
}
