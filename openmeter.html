func validateVersion(version string) (bool, error) {
    semverPattern := regexp.MustCompile(`^v?\d+\.\d+\.\d+(-[\w.]+)?(\+[\w.]+)?$`)
    if !semverPattern.MatchString(version) {
        return false, fmt.Errorf("invalid version format: %s", version)
    }
    return true, nil
}
