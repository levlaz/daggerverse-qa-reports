2. Add Error Handling and Retries

Kafka connectivity tests should include proper retry mechanisms.

// Example implementation:
func clusterConnect() *Test {
    maxRetries := 3
    retryDelay := time.Second * 5
    
    for i := 0; i < maxRetries; i++ {
        if err := tryConnect(); err != nil {
            if i == maxRetries - 1 {
                return &Test{Success: false, Error: err}
            }
            time.Sleep(retryDelay)
            continue
        }
        return &Test{Success: true}
    }
}
