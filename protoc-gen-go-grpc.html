func (m *Module) runWithRetry(ctx context.Context, cmd []string) error {
    maxRetries := 3
    for i := 0; i < maxRetries; i++ {
        err := m.run(ctx, cmd)
        if err == nil {
            return nil
        }
        if i < maxRetries-1 {
            time.Sleep(time.Second * time.Duration(i+1))
            continue
        }
        return fmt.Errorf("failed after %d retries: %w", maxRetries, err)
    }
    return nil
}
