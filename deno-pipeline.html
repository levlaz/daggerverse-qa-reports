Suggestions for Improvement

1. Add Input Validation

Add proper input validation for the pipeline parameters to prevent errors during runtime.

// Example implementation
export function validateInput(options: PipelineOptions): Result {
    if (!options.src) {
        return { error: new Error("Source directory is required") };
    }
    if (options.buildFlags && !Array.isArray(options.buildFlags)) {
        return { error: new Error("Build flags must be an array") };
    }
    return { value: options };
}
