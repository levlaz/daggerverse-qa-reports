Module Description

This module provides functionality for running Ruff (Python linter) checks and tests. It includes several functions for different testing scenarios.

Available Functions

  • all() - All runs all tests.
  • check() - Check runs the ruff check command.
  • check-with-config() - CheckWithConfig runs the ruff check command with a configuration file.

func TestCheck(t *testing.T) {
    tests := []struct {
        name    string
        input   string
        want    bool
        wantErr bool
    }{
        {
            name:    "valid case",
            input:   "test_data",
            want:    true,
            wantErr: false,
        },
    }
    
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            got, err := Check(tt.input)
            if (err != nil) != tt.wantErr {
                t.Errorf("Check() error = %v, wantErr %v", err, tt.wantErr)
                return
            }
            if got != tt.want {
                t.Errorf("Check() = %v, want %v", got, tt.want)
            }
        })
    }
}
