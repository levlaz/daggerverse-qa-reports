3. Add Command Validation and Sanitization

Implement command validation to prevent unsafe operations.

func validateGhCommand(cmd string) error {
    forbiddenCommands := []string{"delete", "force", "--force"}
    for _, forbidden := range forbiddenCommands {
        if strings.Contains(cmd, forbidden) {
            return fmt.Errorf("command contains forbidden operation: %s", forbidden)
        }
    }
    return nil
}
