3. Add Multi-file Support

Enable checking multiple files in one operation for better performance.

// Example implementation
func (m *Module) checkMultiple(ctx context.Context, files []string) (map[string]*File, error) {
    results := make(map[string]*File)
    for _, file := range files {
        result, err := m.check(ctx, file)
        if err != nil {
            results[file] = &File{Error: err.Error()}
            continue
        }
        results[file] = result
    }
    return results, nil
}
