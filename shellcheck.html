3. Add Better Error Reporting

Enhance error reporting with more detailed information and formatting.


type ShellcheckResult struct {
    File     string   `json:"file"`
    Line     int      `json:"line"`
    Column   int      `json:"col"`
    Level    string   `json:"level"`
    Code     int      `json:"code"`
    Message  string   `json:"message"`
}

func formatResults(results []ShellcheckResult) string {
    var output strings.Builder
    for _, result := range results {
        fmt.Fprintf(&output, "%s:%d:%d [SC%d] %s\n",
            result.File, result.Line, result.Column,
            result.Code, result.Message)
    }
    return output.String()
}
