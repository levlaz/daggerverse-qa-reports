3. Add Batch Processing Capability

Add the ability to check multiple files at once.

func (m *Module) checkMultiple(ctx context.Context, paths []string) (map[string]*checkResult, error) {
    results := make(map[string]*checkResult)
    for _, path := range paths {
        result, err := m.check(ctx, path)
        if err != nil {
            results[path] = &checkResult{Error: err.Error()}
            continue
        }
        results[path] = result
    }
    return results, nil
}
