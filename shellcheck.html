2. Add Configuration Options

type ShellcheckOpts struct {
    Severity string   // error, warning, info, style
    Exclude  []string // Rules to exclude
}

func (s *Shellcheck) WithOptions(opts ShellcheckOpts) *Shellcheck {
    return s.WithExec([]string{
        "shellcheck",
        "-S", opts.Severity,
        "--exclude", strings.Join(opts.Exclude, ","),
    })
}

func (s *Shellcheck) CheckMultiple(ctx context.Context, scripts []*File) (*File, error) {
